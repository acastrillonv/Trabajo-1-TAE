library(readr)
library(kableExtra)
CollegeScorecard <- read_csv("CollegeScorecard.csv",show_col_types = FALSE)
CollegeScorecardDataDictionary_09_12_2015 <- read_csv("CollegeScorecardDataDictionary-09-12-2015.csv",show_col_types = FALSE)
CollegeScorecard$CURROPER<- as.factor(CollegeScorecard$CURROPER)
CollegeScorecard<-subset(CollegeScorecard, CURROPER!="0")
data<-subset(CollegeScorecard, select=c("INSTNM","CONTROL","ADM_RATE","ADM_RATE_ALL","COSTT4_A","COSTT4_P","PCTPELL","PCTFLOAN"))
data$CONTROL<-as.character(data$CONTROL)
data["CONTROL"][data["CONTROL"] == 1] <- "PUBLIC"
data["CONTROL"][data["CONTROL"] == 2] <- "PRIVATE NONPROFIT"
data["CONTROL"][data["CONTROL"] == 3] <- "PRIVATE FOR-PROFIT"
data$CONTROL<-as.factor(data$CONTROL)
kable(summary(data),caption = "summary") %>% 
  kable_paper("hover", 
              full_width = F) %>% 
  kable_minimal()
